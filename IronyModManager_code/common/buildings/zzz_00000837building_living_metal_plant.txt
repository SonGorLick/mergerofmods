building_living_metal_plant = {
    base_buildtime = @b2_time
    category = manufacturing
    potential = {
        NOT = {
            has_modifier = resort_colony
        }
        NOT = {
            has_modifier = slave_colony
        }
        building_basic_income_check = yes
    }
    allow = {
        has_upgraded_capital = yes
    }
    destroy_trigger = {
        OR = {
            buildings_no_motes = yes
            AND = {
                free_building_slots = 0
                exists = owner
                owner = {
                    is_ai = yes
                    has_monthly_income = {
                        resource = sr_living_metal
                        value > 9
                    }
                }
            }
        }
    }
    triggered_planet_modifier = {
        potential = {
            exists = owner
            owner = {
                is_gestalt = no
            }
        }
        modifier = {
            job_living_metal_farmer_add = 1
        }
    }
    triggered_planet_modifier = {
        potential = {
            exists = owner
            owner = {
                is_gestalt = yes
            }
        }
        modifier = {
            job_living_metal_farmer_drone_add = 1
        }
    }
    resources = {
        category = planet_buildings
        cost = {
            minerals = @b2_minerals
            volatile_motes = 40
            sr_living_metal = 1
        }
        upkeep = {
            energy = 3
        }
    }
    prerequisites = {
        "tech_living_metal_plant"
    }
    triggered_desc = {
        text = job_living_metal_farmer_effect_desc
    }
    ai_resource_production = {
        sr_living_metal = 1
        trigger = {
            always = yes
        }
    }
    on_queued = {
        IF = {
            limit = {
                owner = {
                    has_technology = tech_construction_templates
                }
            }
            owner = {
                set_timed_country_flag = {
                    flag = buildings_new_living_metal_plant
                    days = 270
                }
            }
        }
        ELSE_IF = {
            limit = {
                owner = {
                    has_technology = tech_assembly_pattern
                }
            }
            owner = {
                set_timed_country_flag = {
                    flag = buildings_new_living_metal_plant
                    days = 390
                }
            }
        }
        ELSE = {
            planet = {
                owner = {
                    set_timed_country_flag = {
                        flag = buildings_new_living_metal_plant
                        days = 510
                    }
                }
            }
        }
    }
}
